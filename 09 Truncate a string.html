基本思路

function truncate(str, num) {
    if (str.length > num) {
        if (num > 3) {
            // num 大于 3，因此要少截三位，留出 ... 的位置
            return str.substr(0, num - 3) + '...';
        } else {
            // num 小于 3，正常截取，并把 ... 补到后面
            return str.substr(0, num) + '...';
        }
    } else {
        // num 比字符串长度大
        return str;
    }
}

优化思路

function truncate(str, num) {
    if (str.length > num) {
        return str.substr(0, num > 3 ? num - 3 : num) + '...';
    }
    return str;
}
