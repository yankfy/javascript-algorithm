基本思路

function repeat(str, num) {
    var result = "";
    for (var i = 0; i < num; i++) {
        result += str;
    }
    return result;
}

优化思路

function repeat(str, num) {
    var result = "";
    while (num > 0) {
        result += str;
        num--;
    }
    return result;
}

使用repeat()

function repeat(str, num) {
    return num > 0 ? str.repeat(num) : "";
}

使用join()

function repeat(str, num) {
    if (num > 0) {
        return Array(num + 1).join(str);
    }
    return "";
}
