function confirmEnding(str, target) {
    // 计算出开始的索引
    var startIndex = str.length - target.length;
    // 从开始索引，一直截取到字符串终点
    var strEnd = str.slice(startIndex);
    return strEnd === target;
}

优化思路

function confirmEnding(str, target) {
    return str.slice(-target.length) === target;
}

利用正则来解决

function confirmEnding(str, target) {
    return RegExp(target + '$').test(str);
}
