基本思路

function sumAll(arr) {
    // 用 left 代表较小的数，right 代表较大的数
    var left, right;
    var sum = 0;
    if (arr[0] < arr[1]) {
        left = arr[0];
        right = arr[1];
    } else {
        left = arr[1];
        right = arr[0];
    }
    for (var i = left; i <= right; i++) {
        sum += i;
    }
    return sum;
}

优化思路 等差數列

function sumAll(arr) {
    var left = Math.min.apply(null, arr);
    var right = Math.max.apply(null, arr);
    return (left + right) * (right - left + 1) / 2;
}

优化 使用数组方法

function sumAll(arr) {
    var left = Math.min.apply(null, arr);
    var right = Math.max.apply(null, arr);
    var numArr = [];
    for (var i = left; i <= right; i++) {
        numArr.push(i);
    }
    return numArr.reduce(function(prev, next) {
        return prev + next;
    })
}

优化 使用Array.from



function sumAll(arr) {
    var numArr = Array.from({
            length: Math.abs(arr[0] - arr[1]) + 1
        }, (_, i) => 
            return i + Math.min.apply(null, arr)
    )
    return numArr.reduce((prev, next) => prev + next);
}
